# Mini source/target extension example

This example is documentation-only for now: dynamic plugin loading is not part of the public preview. Use it as a shape guide when contributing a new built-in frontend/backend or embedding Migrator.Core in your own host.

## Mini source frontend

```csharp
using Migrator.Core;
using Migrator.Core.Models.Ir;
using Migrator.Core.SourceFrontends;

public sealed class MiniSeleniumFrontend : ISourceFrontend
{
    public SourceSpec Source { get; } = new("selenium-mini", "mini", "selenium");
    public IReadOnlyCollection<string> Aliases { get; } = new[] { "mini" };

    public SourceCapabilityReport Capabilities { get; } = new(
        SchemaVersion: "source-capabilities/v1",
        Source: new SourceSpec("selenium-mini", "mini", "selenium"),
        Status: "experimental-spike",
        Summary: "Minimal example frontend for extension authors.",
        Capabilities: new[]
        {
            new SourceCapabilityItem("semantic-model", "none", "Example only; no semantic model.", Array.Empty<string>()),
            new SourceCapabilityItem("locators", "basic", "Recognizes a tiny selector subset.", new[] { "find('#save')" })
        },
        Limitations: new[] { "Documentation sample only." },
        RecommendedValidation: new[] { "Add real fixture tests before use." });

    public bool CanParse(MigrationRequest request) =>
        request.Source.Id.Equals(Source.Id, StringComparison.OrdinalIgnoreCase)
        || request.InputPath.EndsWith(".mini", StringComparison.OrdinalIgnoreCase);

    public SourceParseResult Parse(MigrationRequest request)
    {
        // Real frontends should parse source files and return MigrationDocument IR V2.
        return new SourceParseResult(Array.Empty<MigrationDocument>(), Array.Empty<IrDiagnostic>());
    }
}
```

## Mini target backend

```csharp
using Migrator.Core;
using Migrator.Core.Models;
using Migrator.Core.Models.Ir;

public sealed class MiniTargetBackend : ITargetBackend
{
    public TargetSpec Target { get; } = new("playwright-mini", "mini", "playwright");
    public IReadOnlyCollection<string> Aliases { get; } = new[] { "mini-target" };

    public TargetCapabilityReport Capabilities => new(
        SchemaVersion: "target-capabilities/v1",
        Target: Target,
        Status: "experimental-spike",
        Summary: "Minimal example backend for extension authors.",
        Capabilities: new[]
        {
            new TargetCapabilityItem("ir-v2-rendering", "limited", "Example renderer only.", Array.Empty<string>())
        },
        Limitations: new[] { "Documentation sample only." },
        RecommendedValidation: new[] { "Add renderer snapshots and project verification before use." });

    public string Render(TestFileModel model) => "// generated by mini backend";
    public string RenderDocument(MigrationDocument document) => "// generated from IR V2 by mini backend";
    public string GetDefaultFileName(TestFileModel model) => $"{model.ClassName}.mini";
}
```

## Registration shape

```csharp
var sourceRegistry = new SourceFrontendRegistry()
    .Register(new MiniSeleniumFrontend());

var targetRegistry = new TargetBackendRegistry()
    .Register(new MiniTargetBackend());
```

A production contribution should also update `--mode capabilities`, docs, schemas if needed, fixture tests, and CLI smoke tests.
